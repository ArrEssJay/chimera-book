  (see {[}{[}Bibliography{]}{]})

\begin{itemize}

\item
  Add to {[}{[}Home{]}{]} in appropriate Part (I-VIII)
\item
  Update {[}{[}Bibliography{]}{]} if introducing new references
\end{itemize}
